  241.      *          character set conversion for both input and output values.

  247.     public function __call($methodName, $arguments)
  248.     {
  249.         if ($this->_compatibleMode 
  250.             && array_key_exists($methodName, $this->_compatible)
  251.         ) {
  252.             $methodName = $this->_compatible[$methodName];
  253.         }
  254.  
  255.         // setMode & getMode [Date*|Query], setLang [Soundex*|CompressStr]  
  256.         if ('I18N_Arabic_'.$this->_lazyLoading[$methodName] != $this->myClass) {
  257.             $this->load($this->_lazyLoading[$methodName]);
  258.         }
  259.  
  260.         // Create an instance of the ReflectionMethod class
  261.         $method = new ReflectionMethod($this->myClass, $methodName);
  262.         
  263.         $params     = array();
  264.         $parameters = $method->getParameters();
  265.  
  266.         foreach ($parameters as $parameter) {
  267.             $name  = $parameter->getName();
  268.             $value = array_shift($arguments);
  269.             
  270.             if (is_null($value) && $parameter->isDefaultValueAvailable()) {
  271.                 $value = $parameter->getDefaultValue();
  272.             }
  273.             
  274.             if ($methodName == 'decompress'
  275.                 || ($methodName == 'search' && $name == 'bin')
  276.                 || ($methodName == 'length' && $name == 'bin')
  277.             ) {
  278.                 $params[$name] = $value;
  279.             } else {
  280.                 $params[$name] = iconv($this->getInputCharset(), 'utf-8', $value);
  281.             }
  282.         }
  283.  
  284.         $value = call_user_func_array(array(&$this->myObject, $methodName), $params);
  285.  
  286.         if ($methodName == 'tagText') {
  287.             $outputCharset = $this->getOutputCharset();
  288.             foreach ($value as $key=>$text) {
  289.                 $value[$key][0] = iconv('utf-8', $outputCharset, $text[0]);
  290.             }
  291.         } elseif ($methodName == 'compress' 
  292.                   || $methodName == 'getPrayTime'
  293.                   || $methodName == 'str2graph'
  294.         ) {
  295.         } else {
  296.             $value = iconv('utf-8', $this->getOutputCharset(), $value);
  297.         }
  298.  
  299.         return $value;
  300.     }

  431.     {

  433.         $head = '';
  434.
