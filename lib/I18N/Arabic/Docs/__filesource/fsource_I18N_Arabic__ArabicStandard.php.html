  88.      * This method will standardize Arabic text to follow writing standards
  89.      * (just like magazine rules)
  90.      *          
  91.      * @param string $text Arabic text you would like to standardize
  92.      *                    
  93.      * @return String Standardized version of input Arabic text
  94.      * @author Khaled Al-Sham'aa <khaled@ar-php.org>
  95.      */
  96.     public static function standard($text)
  97.     {
  98.         $patterns     = array();
  99.         $replacements = array();
  100.         
  101.         array_push($patterns, '/\r\n/u', '/([^\@])\n([^\@])/u', '/\r/u');
  102.         array_push($replacements, "\n@@@\n", "\\1\n&&&\n\\2", "\n###\n");
  103.         
  104.         /**
  105.          * النقطة، الفاصلة، الفاصلة المنقوطة،
  106.          * النقطتان، علامتي الاستفهام والتعجب،
  107.          * النقاط الثلاث المتتالية
  108.          * يترك فراغ واحد بعدها جميعا
  109.          * دون أي فراغ قبلها
  110.          */
  111.         array_push($patterns, '/\s*([\.\،\؛\:\!\؟])\s*/u');
  112.         array_push($replacements, '\\1 ');
  113.  
  114.         /**
  115.          * النقاط المتتالية عددها 3 فقط
  116.          * (ليست نقطتان وليست أربع أو أكثر)
  117.          */
  118.         array_push($patterns, '/(\. ){2,}/u');
  119.         array_push($replacements, '...');
  120.  
  121.         /**
  122.          * الأقواس ( ) [ ] { } يترك قبلها وبعدها فراغ
  123.          * وحيد، فيما لا يوجد بينها وبين ما بداخلها
  124.          * أي فراغ
  125.          */
  126.         array_push($patterns, '/\s*([\(\{\[])\s*/u');
  127.         array_push($replacements, ' \\1');
  128.  
  129.         array_push($patterns, '/\s*([\)\}\]])\s*/u');
  130.         array_push($replacements, '\\1 ');
  131.  
  132.         /**
  133.          * علامات الاقتباس "..."
  134.          * يترك قبلها وبعدها فراغ
  135.          * وحيد، فيما لا يوجد بينها
  136.          * وبين ما بداخلها أي فراغ
  137.          */
  138.         array_push($patterns, '/\s*\"\s*(.+)((?<!\s)\"|\s+\")\s*/u');
  139.         array_push($replacements, ' "\\1" ');
  140.  
  141.         /**
  142.          * علامات الإعتراض -...-
  143.          * يترك قبلها وبعدها فراغ
  144.          * وحيد، فيما لا يوجد بينها
  145.          * وبين ما بداخلها أي فراغ
  146.          */
  147.         array_push($patterns, '/\s*\-\s*(.+)((?<!\s)\-|\s+\-)\s*/u');
  148.         array_push($replacements, ' -\\1- ');
  149.  
  150.         /**
  151.          * لا يترك فراغ بين حرف العطف الواو وبين
  152.          * الكلمة التي تليه
  153.          * إلا إن كانت تبدأ بحرف الواو
  154.          */
  155.         array_push($patterns, '/\sو\s+([^و])/u');
  156.         array_push($replacements, ' و\\1');
  157.  
  158.         /**
  159.          * الواحدات الإنجليزية توضع
  160.          * على يمين الرقم مع ترك فراغ
  161.          */
  162.         array_push($patterns, '/\s+(\w+)\s*(\d+)\s+/');
  163.         array_push($replacements, ' <span dir="ltr">\\2 \\1</span> ');
  164.  
  165.         array_push($patterns, '/\s+(\d+)\s*(\w+)\s+/');
  166.         array_push($replacements, ' <span dir="ltr">\\1 \\2</span> ');
  167.  
  168.         /**
  169.          * النسبة المؤية دائما إلى يسار الرقم
  170.          * وبدون أي فراغ يفصل بينهما 40% مثلا
  171.          */
  172.         array_push($patterns, '/\s+(\d+)\s*\%\s+/u');
  173.         array_push($replacements, ' %\\1 ');
  174.         
  175.         array_push($patterns, '/\n?@@@\n?/u', '/\n?&&&\n?/u', '/\n?###\n?/u');
  176.         array_push($replacements, "\r\n", "\n", "\r");
  177.  
  178.         $text = preg_replace($patterns, $replacements, $text);
  179.  
  180.         return $text;
  181.     }
