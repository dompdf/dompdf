  309.     /**
  310.      * Regression analysis calculate roughly the max number of character fit in
  311.      * one A4 page line for a given font size.
  312.      *      
  313.      * @param integer $font Font size
  314.      *      
  315.      * @return integer Maximum number of characters per line
  316.      * @author Khaled Al-Sham'aa <khaled@ar-php.org>
  317.      */
  318.     public function a4MaxChars($font)
  319.     {
  320.         $x = 381.6 - 31.57 * $font + 1.182 * pow($font, 2) - 0.02052 * 
  321.              pow($font, 3) + 0.0001342 * pow($font, 4);
  322.         return floor($x - 2);
  323.     }

  325.     /**
  326.      * Calculate the lines number of given Arabic text and font size that will
  327.      * fit in A4 page size
  328.      *      
  329.      * @param string  $str  Arabic string you would like to split it into lines
  330.      * @param integer $font Font size
  331.      *                    
  332.      * @return integer Number of lines for a given Arabic string in A4 page size
  333.      * @author Khaled Al-Sham'aa <khaled@ar-php.org>
  334.      */
  335.     public function a4Lines($str, $font)
  336.     {
  337.         $str = str_replace(array("\r\n", "\n", "\r"), "\n", $str);
  338.         
  339.         $lines     = 0;
  340.         $chars     = 0;
  341.         $words     = explode(' ', $str);
  342.         $w_count   = count($words);
  343.         $max_chars = $this->a4MaxChars($font);
  344.         
  345.         for ($i = 0; $i < $w_count; $i++) {
  346.             $w_len = mb_strlen($words[$i]) + 1;
  347.             
  348.             if ($chars + $w_len < $max_chars) {
  349.                 if (mb_strpos($words[$i], "\n") !== false) {
  350.                     $words_nl = explode("\n", $words[$i]);
  351.                     
  352.                     $nl_num = count($words_nl) - 1;
  353.                     for ($j = 1; $j < $nl_num; $j++) {
  354.                         $lines++;
  355.                     }
  356.                     
  357.                     $chars = mb_strlen($words_nl[$nl_num]) + 1;
  358.                 } else {
  359.                     $chars += $w_len;
  360.                 }
  361.             } else {
  362.                 $lines++;
  363.                 $chars = $w_len;
  364.             }
  365.         }
  366.         $lines++;
  367.         
  368.         return $lines;
  369.     }
